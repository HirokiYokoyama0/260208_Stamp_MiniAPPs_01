#### 6. `app/api/stamps/scan/route.ts` (アプリ内スキャン用)
**変更内容**:
1. インターフェース型に`purchase`追加

3. 1日1回制限を`type !== "purchase"`で条件分岐
4. stamp_historyへの挿入で条件分岐

```typescript
// 1. インターフェース更新
interface QRScanRequest {
  userId: string;
  type: "premium" | "regular" | "purchase";  // ← purchase追加
  stamps: number;
  qrCodeId?: string;
}

// 2. バリデーション更新
if (type !== "premium" && type !== "regular" && type !== "purchase") {
  return NextResponse.json({ ... }, { status: 400 });
}

// 3. 1日1回制限を条件分岐
if (type !== "purchase") {
  // Premium/Regular のみ1日1回制限チェック
  const { data: todayQrRecords } = await supabase
    .from("stamp_history")
    .eq("stamp_method", "qr")
    .gte("visit_date", `${today}T00:00:00`)
    .lt("visit_date", `${today}T23:59:59.999Z`);

  if (todayQrRecords && todayQrRecords.length > 0) {
    return NextResponse.json({ ... }, { status: 409 });
  }
}

// 4. stamp_history挿入
const { data: stampData } = await supabase
  .from("stamp_history")
  .insert({
    user_id: userId,
    stamp_method: type === "purchase" ? "purchase_incentive" : "qr",  // ← 条件分岐
    amount: stamps,
    notes: type === "purchase"
      ? "購買インセンティブ (アプリ内スキャン)"
      : `アプリ内スキャン (${type})`,
  });
```

**内容**: stamp_historyテーブルのCHECK制約を更新

```sql
-- 既存のCHECK制約を削除
ALTER TABLE stamp_history
DROP CONSTRAINT IF EXISTS stamp_history_stamp_method_check;

-- 新しいCHECK制約を追加（purchase_incentiveを含める）
ALTER TABLE stamp_history
ADD CONSTRAINT stamp_history_stamp_method_check
CHECK (stamp_method IN (
  'qr',                 -- 現在のQR読み取り実装

  'slot_game',          -- スロットゲーム
  'purchase_incentive'  -- 購買インセンティブQR読み取り（新規追加）
));
```
